calculate_word_occurrence <- function(input_file, output_file) {
  # Read the contents of the input file
  text <- tolower(readLines(input_file))
  
  # Combine all lines into a single character vector
  text <- paste(text, collapse = " ")
  
  # Remove punctuation
  text <- gsub("[[:punct:]]", "", text)
  
  # Split the text into individual words
  words <- unlist(strsplit(text, "\\s+"))
  
  # Remove empty strings
  words <- words[words != ""]
  
  # Calculate word frequencies
  word_counts <- table(words)
  
  # Sort word frequencies in ascending order
  word_counts <- sort(word_counts)
  
  # Write word frequencies to output file
  write.table(word_counts, file = output_file, quote = FALSE, col.names = FALSE)
}

# Example usage
calculate_word_occurrence("input.txt", "output.txt")

calculate_word_occurrence <- function(input_file, output_file) {
  # Read the contents of the input file
  text <- tolower(readLines(input_file))
  
  # Combine all lines into a single character vector
  text <- paste(text, collapse = " ")
  
  # Remove punctuation
  text <- gsub("[[:punct:]]", "", text)
  
  # Split the text into individual words
  words <- unlist(strsplit(text, "\\s+"))
  
  # Remove empty strings and numeric values
  words <- words[words != "" & !grepl("^\\d+$", words)]
  
  # Calculate word frequencies
  word_counts <- table(words)
  
  # Sort word frequencies in ascending order
  word_counts <- sort(word_counts)
  
  # Write word frequencies to output file
  write.table(word_counts, file = output_file, quote = FALSE, col.names = FALSE)
}

# Example usage
calculate_word_occurrence("input.txt", "output.txt")

# Load required library
library(wordcloud)

plot_word_cloud <- function(output_file) {
  # Read word frequencies from the output file
  word_freq <- read.table(output_file, header = FALSE, col.names = c("Word", "Frequency"))
  
  # Create word cloud
  wordcloud(words = word_freq$Word, freq = word_freq$Frequency, min.freq = 1,
            scale = c(4, 0.5), random.order = FALSE,
            colors = brewer.pal(8, "Dark2"))
}

# Example usage
plot_word_cloud("output.txt")
